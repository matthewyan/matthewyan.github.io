JS

let express = require("express");
let app = express();

 // 加载指定目录作为静态页面目录
app.use(express.static("../develop/_book"));

app.listen(8080, ()=>{
    console.log("listen 8080");
})

GO

package main

import (
    "log"
    "net/http"
)

func main() {
    // 使用_book作为静态页面目录
    h := http.FileServer(http.Dir("./_book"))
    err := http.ListenAndServe(":9090", h)
    if err != nil {
        log.Fatal("ListenAndServe: ", err)
    }
}
